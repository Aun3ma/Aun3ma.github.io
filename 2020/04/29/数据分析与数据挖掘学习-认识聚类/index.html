Q1

Q1比较简单，对于数据做必要处理后直接使用K-Means即可，唯一的问题是大部分的库中的K-Means算法不支持自定义距离函数或者是自定义比较麻烦，所以我们可以选择直接自己写一个简单的K-Means函数。

数据处理

选取需要用到的数据（vipno、pluno、atm等等），并按照vipno进行划分，最后计算每个商品种类的总额。

# 将数据集按照用户进行划分，返回一个dataframe的集合
def group_by_vipno(this_data_frame):
    data_frame = this_data_frame.sort_values(axis=0, ascending=True, by='vipno')
    data_set = []
    data_frame = data_frame.reset_index(drop=True)
    last_vipno = data_frame.loc[0, 'vipno']
    last_index = 0
    i = 0
    while i < data_frame.shape[0]:
        if data_frame.loc[i, 'vipno'] == last_vipno:
            i = i + 1
            if i == data_frame.shape[0]:
                data = data_frame[last_index:i]
                data_set.append(data)
            continue
        else:
            data = data_frame[last_index:i]
            data_set.append(data)
            last_vipno = data_frame.loc[i, 'vipno']
            last_index = i
            i = i + 1
    return data_set

# 计算每个用户购买每种商品的总额，返回一个dataframe的集合
def count_total_money(this_data_set):

Jaccard相似度计算

# 计算两个用户的dataframe之间的jaccard相似度，返回Jaccard相似度数值
def count_jaccard_coefficient(this_user_info_one, this_user_info_two):
    union_info = np.maximum(this_user_info_two, this_user_info_one)
    intersec_info = np.minimum(this_user_info_two, this_user_info_one)
    union = np.sum(union_info)
    intersection = np.sum(intersec_info)
    jaccard_coefficient = intersection / union
    return jaccard_coefficient

求解簇中心点

用于K-Means算法求中心点，将一个簇内的所有种类商品总额求和取平均。

# 寻找簇的中心点，返回中心点的dataframe
def find_center_point(this_array_set, index):
    union_data_array = np.zeros(40000)
    for i in index:
        union_data_array = union_data_array + this_array_set[i]
    union_data_array = union_data_array / len(index)
    return union_data_array

简单K-Means算法

最简单的K-Means算法，随机选择初始的中心点，每次聚类循环30次。

# 简单K-Means算法，每次聚类循环30次,返回聚类结果
def my_k_means(this_array_set, k_count):
    step = 0
    count = len(this_array_set)
    k = rd.sample(range(count), k_count)
    k_point = []
    k.sort()
    for i in k:
        k_point.append(this_array_set[i])
    while True:
        step = step + 1
        km = [[] for i in range(k_count)]
        for i in range(count):
            all_jaccard = []
            for j in range(k_count):
                jaccard = count_jaccard_coefficient(this_array_set[i], k_point[j])
                all_jaccard.append(jaccard)
            max_index = all_jaccard.index(max(all_jaccard))
            km[max_index].append(i)
        k_new = []
        for i in km:
            new_center_point_array = find_center_point(this_array_set, i)
            k_new.append(new_center_point_array)
        if step < 30:
            k_point = k_new
        else:
            km_label = np.zeros(count)
            cluster_num = 0
            for i in km:
                cluster_num = cluster_num + 1
                for j in i:
                    km_label[j] = cluster_num
            evaluate_cluster(this_array_set, km_label, k_count, km, k_point)
            break
    return km

Jaccard相似度计算

由于分成了四级，所以要求四级Jaccard相似度后取平均。

# 计算两个用户的dataframe之间的jaccard相似度的平均值，返回Jaccard相似度的平均值的数值
def count_average_jaccard_coefficient(this_user_info_one, this_user_info_two):
    total_jaccard = 0
    for lv in range(4):
        if lv == 0:
            total_jaccard = total_jaccard + count_jaccard_coefficient(
                this_user_info_one[:100], this_user_info_two[:100])
        if lv == 1:
            total_jaccard = total_jaccard + count_jaccard_coefficient(
                this_user_info_one[101:1000], this_user_info_two[101:1000])
        if lv == 2:
            total_jaccard = total_jaccard + count_jaccard_coefficient(
                this_user_info_one[1001:10000], this_user_info_two[1001:10000])
        if lv == 3:
            total_jaccard = total_jaccard + count_jaccard_coefficient(
                this_user_info_one[10001:], this_user_info_two[10001:])
    average_jaccard = total_jaccard / 4
    return average_jaccard

求两颗FTC树之间的距离

求距离需要用到两颗树的交树与并树，请大家自行编写方法获取并树与交树。
距离的定义参考提供的材料。

# 计算两颗树之间的FTC距离
def count_ftc_distance(this_ftc_tree, k_ftc_tree):
    my_union_tree = union_tree(this_ftc_tree, k_ftc_tree)
    my_intersection_tree = intersection_tree(this_ftc_tree, k_ftc_tree)
    height = my_intersection_tree.depth() + 1
    wl = 0
    for i in range(height):
        wl = wl + i
    if wl == 0:
        return 1
    node_list = my_intersection_tree.all_nodes()
    total_sim_v1_ur = 0
    total_sim_v2_ur = 0
    total_sim_v3_ur = 0
    total_sim_v4_ur = 0
    count_v1 = 1
    count_v2 = 0
    count_v3 = 0
    count_v4 = 0
    for node in node_list:
        if node.identifier == 'root':
            continue
        lv = len(node.identifier)
        if lv == 2:
            # count_v1 = count_v1 + 1
            if len(my_union_tree.children(node.identifier)) != 0:
                count_v2 = count_v2 + 1
            divide_num = 0
            child_node_list = my_union_tree.children(my_union_tree.parent(node.identifier).identifier)
            for c_node in child_node_list:
                divide_num = divide_num + c_node.data
            total_sim_v1_ur = total_sim_v1_ur + (node.data / divide_num)
        if lv == 3:
            # count_v2 = count_v2 + 1
            if len(my_union_tree.children(node.identifier)) != 0:
                count_v3 = count_v3 + 1
            divide_num = 0
            child_node_list = my_union_tree.children(my_union_tree.parent(node.identifier).identifier)
            for c_node in child_node_list:
                divide_num = divide_num + c_node.data
            total_sim_v2_ur = total_sim_v2_ur + (node.data / divide_num)
        if lv == 4:
            # count_v3 = count_v3 + 1
            if len(my_union_tree.children(node.identifier)) != 0:
                count_v4 = count_v4 + 1
            divide_num = 0
            child_node_list = my_union_tree.children(my_union_tree.parent(node.identifier).identifier)
            for c_node in child_node_list:
                divide_num = divide_num + c_node.data
            total_sim_v3_ur = total_sim_v3_ur + (node.data / divide_num)
        if lv == 5:
            # count_v4 = count_v4 + 1
            divide_num = 0
            child_node_list = my_union_tree.children(my_union_tree.parent(node.identifier).identifier)
            for c_node in child_node_list:
                divide_num = divide_num + c_node.data
            total_sim_v4_ur = total_sim_v4_ur + (node.data / divide_num)
    sim_v1_ur = 0
    sim_v2_ur = 0
    sim_v3_ur = 0
    sim_v4_ur = 0
    if count_v1 != 0:
        sim_v1_ur = total_sim_v1_ur / count_v1
    if count_v2 != 0:
        sim_v2_ur = total_sim_v2_ur / count_v2
    if count_v3 != 0:
        sim_v3_ur = total_sim_v3_ur / count_v3
    if count_v4 != 0:
        sim_v4_ur = total_sim_v4_ur / count_v4
    dist = 1 - (sim_v1_ur * 1 / wl + sim_v2_ur * 2 / wl + sim_v3_ur * 3 / wl + sim_v4_ur * 4 / wl)
    return dist

求解质心树

质心树的求解算法伪代码请参考提供的材料。

# 寻找质心树的方法
def find_center_point(this_ftc_tree_set):
    my_union_tree = Tree()
    my_union_tree.create_node('Root', 'root', parent=None, data=0)
    for tree in this_ftc_tree_set:
        my_union_tree = union_tree(my_union_tree, tree)
    mindist = 1000000
    node_list = my_union_tree.all_nodes()
    total_frequent = 0
    max_frequent = 0
    for node in node_list:
        if node.identifier == 'root':
            continue
        total_frequent = total_frequent + node.data
        if max_frequent < node.data:
            max_frequent = node.data
    average_frequent = max_frequent / 100
    frequent = 0
    center_tree = Tree()
    while frequent < (max_frequent * 1 / 2):
        my_union_tree = update_utree(my_union_tree, frequent)
        dist = 0
        for current_tree in this_ftc_tree_set:
            dist = dist + count_ftc_distance(current_tree, my_union_tree)
        if dist < mindist:
            mindist = dist
            center_tree = Tree()
            center_tree.create_node('Root', 'root', parent=None, data=0)
            center_tree = union_tree(center_tree, my_union_tree)
        frequent = frequent + average_frequent
    return center_tree
